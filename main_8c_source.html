288static void run_all(void) {
289 neo6_run();
290
291 console_run();
292
293 datalog_run_handle(datalog_handle);
294
295 button_run_handle_all();
296
297 button_get_pushed_state(lbtn, &lbtn_pushed);
298 button_get_pushed_state(mbtn, &mbtn_pushed);
299 button_get_pushed_state(rbtn, &rbtn_pushed);
300
301 button_get_held_state(mbtn, &mbtn_held);
302
303 if(lbtn_pushed && can_toggle_lbtn) {
304 menu_scroll(current_menu, DOWN);
305
306 refresh = true;
307 can_toggle_lbtn = false;
308 }
309 else if(!lbtn_pushed && !can_toggle_lbtn) {
310 can_toggle_lbtn = true;
311 }
312
313 if(mbtn_pushed && can_toggle_mbtn) {
314 menu_get_selected_option(current_menu, selected_option);
315
316 if(current_menu == main_menu) {
317 if(str_cmp(selected_option, "LOCATION") == true) {
318 menu_set_prev_menu(location_menu, current_menu);
319 current_menu = location_menu;
320 refresh = true;
321 }
322 else if(str_cmp(selected_option, "DATE/TIME") == true) {
323 menu_set_prev_menu(datetime_menu, current_menu);
324 current_menu = datetime_menu;
325 refresh = true;
326 }
327 else if(str_cmp(selected_option, "MOVEMENT") == true) {
328 menu_set_prev_menu(movement_menu, current_menu);
329 current_menu = movement_menu;
330 refresh = true;
331 }
332 else if(str_cmp(selected_option, "ACCURACY") == true) {
333 menu_set_prev_menu(accuracy_menu, current_menu);
334 current_menu = accuracy_menu;
335 refresh = true;
336 }
337 else if(str_cmp(selected_option, "SATELLITES") == true) {
338 menu_set_prev_menu(satellites_menu, current_menu);
339 current_menu = satellites_menu;
340 refresh = true;
341 }
342 }
343
344 can_toggle_mbtn = false;
345 }
346 else if(!mbtn_pushed && !can_toggle_mbtn) {
347 can_toggle_mbtn = true;
348 }
349
350 if(mbtn_held && can_hold_mbtn) {
351 MENU_HANDLE_ts *prev_menu = { 0 };
352
353 menu_get_prev_menu(current_menu, &prev_menu);
354 current_menu = prev_menu;
355 refresh = true;
356
357 can_hold_mbtn = false;
358 }
359 else if(!mbtn_held && !can_hold_mbtn) {
360 can_hold_mbtn = true;
361 }
362
363 if(rbtn_pushed && can_toggle_rbtn) {
364 menu_scroll(current_menu, UP);
365
366 refresh = true;
367 can_toggle_rbtn = false;
368 }
369 else if(!rbtn_pushed && !can_toggle_rbtn) {
370 can_toggle_rbtn = true;
371 }
372
373 if(systick_get_ms() - refreshed_ms >= 1000) {
374 refresh = true;
375 }
376
377 if(refresh) {
378 menu_run_handle(current_menu);
379 refresh = false;
380 refreshed_ms = systick_get_ms();
381 }
382}

391static ERR_te acquire_neo6_location(uint8_t index, char **value_o) {
392 NEO6_INFO_ts *neo6_info = NULL;
393 neo6_get_info(&neo6_info);
394
395 switch(index) {
396 case 0: *value_o = neo6_info->lat; break;
397 case 1: *value_o = neo6_info->lon; break;
398 case 2: *value_o = neo6_info->ort_height; break;
399 case 3: *value_o = neo6_info->geoid_sep; break;
400 }
401
402 return ERR_OK;
403}
